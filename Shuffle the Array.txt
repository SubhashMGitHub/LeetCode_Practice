class Solution {
    public int[] shuffle(int[] nums, int n) {
        
        int[] opArray=new int[nums.length];
        int j=0;
        for(int i=0;i<n;i++)
        {
            opArray[j]=nums[i];
            opArray[j+1]=nums[i+n];
            j=j+2;
        }
        return opArray;
    }
}
